function countNum(){
		var checkedObj = $('tbody tr .number');
		var totalNum = 0;
		checkedObj.each(function(){
			var num = $(this).text();
			num = parseFloat(num);
			totalNum = + totalNum + num;
			$('.numTotal').text(totalNum);
		});
	};
